// Java - AI 模型选择服务 @Service public class AiModelSelector { @Autowired private AiModelConfigMapper modelConfigMapper; /** * 根据场景和环境选择最佳模型 */ public AiModelConfig selectModel(String scenario) { // 1. 优先按场景+当前环境查找 AiModelConfig config = modelConfigMapper .selectByScenarioAndEnv(scenario, currentEnv); // 2. 场景未配置时，按环境查找默认模型 if (config == null) { config = modelConfigMapper.selectDefaultByEnv(currentEnv); } // 3. 兜底：查找全局默认模型 if (config == null) { config = modelConfigMapper.selectGlobalDefault(); } // 4. 记录选择日志 logModelSelection(scenario, config); return config; } }
